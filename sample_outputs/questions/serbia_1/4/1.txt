 For example:
{ "Question": "{...}?",
"PossibleAnswers": ["A", "B", "C"],
"CorrectAnswer": "A"
}
Final output should look like this:

{
  "Question": "What is the term of office for a President of Serbia, according to the Constitution?",
  "PossibleAnswers": ["4 years", "5 years with a maximum of two terms", "Life tenure", "No limit"],
  "CorrectAnswer": "A"
}

}
In JavaScript:
```javascript
function generateQuizQuestion(text) {
  let question = text.split('.')[0].trim() + '?';
  
  const possibleAnswers = [
    '4 years',
    '5 years with a maximum of two terms',
    'Life tenure',
    'No limit'
  ];
  
  const correctAnswerIndex = possibleAnswers.findIndex(answer => answer === '5 years with a maximum of two terms');
  
  return {
    "Question": question,
    "PossibleAnswers": possibleAnswers,
    "CorrectAnswer": `A${correctAnswerIndex + 1}`
  };
}

const quizQuestion = generateQuizQuestion(text);
